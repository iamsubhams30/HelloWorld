import java.io.*;

class GFG {
	public static void main (String[] args) {
	   String name="suubhaam";
	   int count=0;
	   for (int i=0; i<name.length();i++)
	   {
	       char ch=name.charAt(i);
	       ch=Character.toLowerCase(ch);
	       if(ch=='a' || ch=='e' || ch=='i' || ch=='o' || ch=='u'){
	           count++;
	       }
	   }
	   System.out.println(count);
	}
}
